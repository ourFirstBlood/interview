//快速排序
function quickSort(arr) {
    let l = arr.length
    if(l===0 || l===1) return arr
    let leftArr = [], rightArr = [], x = arr[0]
    for(let i=0; i<l; i++) {
        arr[i]<x ? leftArr.push(arr[i]) : rightArr.push(arr[i])
    }
    return quickSort(leftArr).concat(x, quickSort(rightArr))
}
